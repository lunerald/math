	{\bf Fully rational functions}
	\begin{enumerate}[1.]
		%Task 1
		{\bf \item Given is the function $f(x) = \frac{3}{4} x^4 - 4x^3 + 6x^2 $ with $ D_f = \mathbb{R} $ .}
		
		\begin{enumerate}[a)]
			\item Show that the function has exactly one (double) zero.
			\item Determine the type and location of the points with horizontal tangents.
		\end{enumerate}
		
		{\bf \item The following is known from a completely rational function $ f $ of degree 3:}
		
		\begin{itemize}
			\item $G_f$ is point symmetric about the origin
			\item the point $ H(-1|4) $ is a high point
		\end{itemize}
		
		\noindent Calculate the function term of $f$.
		
		{\bf \item Form the derivative function $f'$.}
		
		\begin{enumerate}[a)]
			\item $$ f(x)= 0.5 x^6 - 2.5x + 4 -{2 \over x^2}$$
			\item $$ f(x)= 2 x^2 \cdot \sqrt{x}$$
		\end{enumerate}
		
		{\bf \item The function $f(x)=0.25x^3+3x^2+9x$ is given; \quad $D_f=\mathbb{R}$;}
		\begin{enumerate}[a)]
			\item Calculate what angle the tangent $t$ to the graph $G_f$ at point $B(-4|-4)$ makes with the x-axis!
			\item Check whether the graph of $f$ still has a tangent parallel to $t$.
		\end{enumerate}
		
		{\bf \item Given is the function $ f(x)= -x^2+5x; \quad D_f=\mathbb{R}$;}
		The curve normals $n_1$ and $n_2$ at the points $N_1(0|0)$ and $N_2(5|0)$ of the graph $G_f$ intersect at the point $P$. Find the area of triangle $N_1PN_2$.
		
		{\bf \item The given function $f$ is a normal parabola that opens downward and is shifted upward by 9 units.} The graph of $f$ is denoted by $G_f$.
		\begin{enumerate}[a)]
			\item Find a function term.
			\item A rectangle $ABCD$ is inscribed between the zeros $G_f$, where $A$ and $B$ are on the x-axis, $C$ and $D$ are on $G_f$. Let the x-coordinate of $B$ be $u$. Create a drawing for $u=2$.
			\item Determine $u$ so that the rectangle area becomes maximum.
		\end{enumerate}
	\end{enumerate}
