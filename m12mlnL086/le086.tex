	\noindent{\bf Task 1. e)}\\
	We determine the roots of the derivative
	$$
	\renewcommand{\arraystretch}{2}
	\begin{array}{rcll}
		f'(x)&=&\ln x +1&\stackrel{!}{=}0\\
		-1&=&\ln x &\Rightarrow \quad x=e^{-1} \\
	\end{array}
	$$
	Monotonicity table:
	$$
	\renewcommand{\arraystretch}{2}
	\begin{array}{c|c|c|c|c|c}
		
		& x<e^{-1} & x=e^{-1} & x>e^{-1} \\
		\hline
		\ln x +1 & - & 0 & + \\
		\hline
		\hline
		f'(x) & - & 0 & + \\
		\hline
		G_f & \searrow & \rightarrow & \nearrow \\
		\hline
		& smf & T & sms & H & smf
	\end{array}
	$$
	$$f(e^{-1})=-e^{-1} \quad \Rightarrow \quad \textnormal{low point} \quad T(e^{-1}|-e^{-1})$$
	$G_f$ is strictly monotonically decreasing for $x \in ]0;e^{-1}]$\\
