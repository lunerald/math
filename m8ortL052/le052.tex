	\noindent {\bf task 3. a)}\\
	$$A=\{ P|d(P;M_1)>r_1 \land P| d(P;M_2)<r_2 \} $$
	
	\noindent {\bf task 3. b)}\\
	$$B=\{ P|d(P;g)\ge d(P;h)\} $$
	
	\noindent {\bf task 3. c)}\\
	The angle between $w_1$ and $w_2$ is $\alpha + \beta$.
	For the angles $\alpha$ and $\beta$ the following applies:
	$$
	\renewcommand{\arraystretch}{1.5}
	\begin{array}{rcll}
		2\cdot \alpha + 2 \cdot \beta &=& 180^\circ & |:2 \\
		\alpha + \beta &=& 90^\circ
		
	\end{array}
	$$
	So $w_1$ and $w_2$ form a right angle.
	
	\noindent {\bf Task 4. a)}\\
	$M_1$ are two locus lines, namely the bisectors of $g_1$ and $g_2$:
	$$M_1=\{P|d(P;g_1)=d(P;g_2)\}$$
	
	\noindent {\bf task 4. b)}\\
	$M_2$ is a local line, the central parallel between $h_1$ and $h_2$:
	$$M_2=\{P|d(P;h_1)=d(P;h_2)\}$$
	
	\noindent {\bf task 4. b)}\\
	$K$ is a spatial region, namely the interior of the circle $k(M;r)$ around $M$ with radius $r$:
	$$K=\{P|d(P;M)\le r\}$$
