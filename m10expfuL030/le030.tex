	\noindent{\bf Task 1. a)}\\
	
	\begin{itemize}
		\item $b$ is the starting value, it can be any real number. We have $b=f(0)$.
		\item $a$ is the growth factor. It must be positive and different from 1:
		$a \in \mathbb{R}^+ \setminus \{1\}.$
	\end{itemize}
	
	\noindent{\bf Task 1. b)}\\
	\begin{itemize}
		\item \underline{Case 1. $a < 0$; e.g. B. $a=-1; \quad x={1\over 2}$}
		$$y=b \cdot a^{1\over 2} = b \cdot \sqrt{-1} \qquad \textnormal{no solution}$$
		\item \underline{Case 2. $a = 0; \quad x=-1$ }
		$$y=b \cdot a^{-1} = b \cdot { 1 \over 0} \qquad \textnormal{no solution}$$
		\item \underline{Case 2. $a = 1$: }
		$$y= b \cdot 1^{x} = b \qquad \textnormal{for all } \qquad x \in \mathbb{R}$$
	\end{itemize}
	
	\noindent{\bf Task 1. c)}\\
	\begin{enumerate}[(i)]
		\item $$f(x)= 2\cdot \left({1\over 2}\right)^x$$
		The function is decreasing because $a<1$ and the exponent is positive. We have $f(0)=2 \not = 1$.
		\item $$f(x)= 2 \cdot 2^x $$
		The function is increasing because $a>1$ and the exponent is positive. We have $f(0)=2$.
	\end{enumerate}

	\noindent{\bf Task 3.}\\
	We plug in the points and get two equations:
	$$
	\renewcommand{\arraystretch}{2}
	\begin{array}{rccll}
		I & 36 &=& b \cdot a^2 \\
		II & {32 \over 3} &=& b \cdot a^5 \\
		\hline
		II : I & {8 \over 27} &=& a^3 \\
		& a &=& {2 \over 3} \\
		& b &=& {36 \cdot 9 \over 4} &= 81\\
		\Rightarrow & f(x)&=& 81 \cdot \left({2 \over 3} \right)^x
	\end{array}
	$$
