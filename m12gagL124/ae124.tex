	{\bf Mathematics -- straight lines in $\mathbb{R}^3$}\\
	\begin{enumerate}[1.]
		{\bf \item Task}\\
		\vspace{0.25 cm}\\
		The line g contains the points $A(2|0|-1)$ and $B(4|5|-3)$.
		\begin{enumerate}[a)]
			\item Decide whether $C(-2|-10|3)$ belongs to the line g.
			\item Decide whether $D(-4|15|5)$ is a point on the line g.
		\end{enumerate}
		{\bf \item Task}\\
		\vspace{0.25 cm}\\
		On the line connecting the points $A(3|6|9)$ and $B(5|7|6)$ lies $C(?|?|4,5)$. You decide
		the coordinates of C.
		\vspace{0.25 cm}\\
		{\bf \item Task} \\
		\vspace{0.25 cm}\\
		The line g contains the points $A(2|0|-3)$ and $B(5|-2|-1)$. The line h contains the point
		$C(4|3|0)$ and is parallel to the line g. Decide whether the point $D(-2|7|-4)$ belongs to the line h
		heard.
		\vspace{0.25 cm}\\
		{\bf \item Task}\\
		\vspace{0.25 cm}\\
		The points $A(1|2|3)$, $B(4|b|4)$ and $C(7|6|c)$ lie on a line g. Determine b and c.
		Give a parametric representation of the line g and use it to decide whether
		$P(-2|0|1)$ belongs to the line g.
		\vspace{0.25 cm}\\
		{\bf \item Task}\\
		\vspace{0.25 cm}\\
		The line g parallel to the y-axis passes through $P(1|3|2)$. Determine a and c so that the point
		$Q(a|5|c)$ belongs to the line g.
		\vspace{0.25 cm}\\
		{\bf \item Task}\\
		\vspace{0.25 cm}\\
		The points $A(6|4|2)$ and $B(8|8|8)$ belong to the line g.
		\begin{enumerate}[a)]
			\item Can c be chosen so that the point $C(3|2|c)$ belongs to g?
			\item Can d be chosen so that $D(3|-2|d)$ belongs to the line g?
			\item Can e be chosen so that
			$\left(
			\begin{array}{c}
				e \\ -3 \\ -4.5\\
			\end{array}
			\right)$
			is a direction vector of g?
		\end{enumerate}
	\end{enumerate}
