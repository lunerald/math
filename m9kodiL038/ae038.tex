	{\bf areas in the coordinate system}
	\begin{enumerate}[1.]
		
		%Task 1
		{\bf \item Calculate the area of the triangle} $ABC$ with $A(-2|3), \\
		B(2.5|-3)$ and $C(6|7.5)$.
		
		{\bf \item The corner points $A(2|3)$ and $C(10|7)$ are known from the parallelogram $AB_nCD_n$.} Points $D_n$ lie on the line $g$ with the equation \\
		$y= - 0.5x + 5$. {\it Note:} The coordinates of $D_n$ are then $D_n(x|-0.5x+5)$
		
		\begin{enumerate}[a)]
			\item Determine the area of the parallelograms depending on $x$.\\
			$[ A(x) = ( -8x + 24 ) ~{\rm AU} ]$
			\item Calculate the coordinates of the point $D_n$ if the parallelogram has an area of 16 AU
		\end{enumerate}
		
		{\bf \item A triangle $ABC$ with $A(2|2)$, $B(x_B|4)$ and $C(10|8)$ has an area of
			19 AU.}\\
		Compute $x_B$ .
		
		{\bf \item Through the points $A(6|3)$, $B_n \in g: \quad y= 2x$ and $C_n \in h:\quad y=-x+3 $ is a family of triangles $AB_nC_n$ set.}
		\begin{enumerate}[a)]
			\item Draw the lines g and h and the triangle $AB_1C_1$ for $x=2$. Space requirement: $-3<x<7$ and $-3<y<7$.
			\item Show that the area $A(x)$ of the triangles depends on the abscissa x:
			$$A(x)=-1.5x^2+10.5x-9 ~{\rm AU}$$
			\item Calculate the points $B_2$ and $C_2$ of the triangle $AB_2C_2$ with the maximum area.
		\end{enumerate}
		
		{\bf \item points $C_n$ on the line } $g:\quad y=-0.4x+4.9$ are corner points of rectangles $OB_nC_nD_n$ whose sides lie on the coordinate axes. The lower left corner of such a rectangle is the origin.
		\begin{enumerate}[a)]
			\item Specify the area of the rectangles $OB_nC_nD_n$ depending on $x$.
			\item For which occupancy of $x$ do we get a rectangle $OB_1C_1D_1$ with maximum area? Enter the maximum area.
			\item For which assignment of $x$ do we get a square? Draw the line and the square $OB_2C_2D_2$ in a coordinate system.
			\item To the right of the square $OB_2C_2D_2$ there should be space for a second square $B_2EC_3F$ under the straight line $g$.
			Calculate the coordinates of $C_3$ and draw this square too.
		\end{enumerate}
		
	\end{enumerate}
