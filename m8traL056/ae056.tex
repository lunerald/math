	{\bf trapezoids}
	\begin{enumerate}[1.]
		
		{\bf \item The four points $P(-1,5|-2)$, $A(x|y)$, $U(5|3)$ and $L(0|3)$ lay the trapezoid $PAUL$ with the parallel sides $[PA]$ and $[UL]$, whose area is $\SI{30}{\centi\meter^2}$.}
		
		\begin{enumerate}[a)]
			\item Draw the points $P$, $U$ and $L$ in a coordinate system (unit 1 cm), determine the coordinates of the point $A$, specify its coordinates and complete the trapezoid.
			\item The parallelogram $PA'U'L$ has an area of $\SI{15}{\centi\meter^2}$ . Calculate the x-coordinates of the points $A'(x|-2)$ and $U'(x|3)$. Draw the parallelogram in blue.
			\item Calculate the fraction of the trapezoid area $PAUL$ that is in II and III. quadrant lies. Express them in percent.
		\end{enumerate}
