	{\bf straight lines, linear functions}
	\begin{enumerate}[1.]
		
		
		
		{\bf \item The line $g$ is defined by the points $A(-3|-1)$ and $B(2|3)$.}
		
		\begin{enumerate}[a)]
			\item Calculate the functional equation of the line $g$. \\
			\item Draw the line $g$ in a coordinate system.\\
			(Length unit $1~cm; \quad -4 \le x \le 5; \quad -5 \le y \le 4 $)
			\item Calculate the missing coordinates of the points $P$ and $Q$ that lie on the line $g$:
			$$ P(x_p|5) \quad Q(-5|y_Q)$$
			\item Check mathematically whether the points $E(-1|0.7)$ and $F(20|17.4)$ lie on the graph of $g$.
			
			\item Give the functional equation of the line $h$ that is orthogonal to $g$ through the point
			$C(0|2)$ runs. Draw the line $h$ in the coordinate system from exercise b).
			
			\item Calculate the functional equation of the line $k$, which is also orthogonal to $g$ and passes through the point $D(-1|1)$. Also draw this line into the coordinate system.
			
			\item Calculate the zero of the line $g$.
			
			\item Create the functional equation of the line $n$ that is parallel to $g$ through the point $Y(0|-1)$. Draw the line $n$ in the coordinate system from b).
			
			\item Draw more lines:
			
			\begin{tabular}{lll}
				t: & $y=-3$ & \\
				s: & $y+5 = 0.6x$ & (Hint: transform to $y=mx+t$)\\
				u: & $y=-1.5(x-1)+2.5$ & (tip: remove the brackets)\\
				v: & $x=1$ &
			\end{tabular}
		\end{enumerate}
		
		{\bf \item Given is the line $g_1$ with $y = 0.4x + 4 $}
		
		\begin{enumerate}[a)]
			\item Draw the line in a coordinate system.
			\item The y-intercept of a line parallel to $g_1$ has the value $1.5$. Draw this line $g_2$ in the same coordinate system and give the corresponding functional equation.
			\item Another parallel line runs through the point $R(-4,5|-5)$. Draw this line $g_3$ in the same coordinate system and determine the corresponding functional equation.
			\item Check mathematically whether the point $A \left( \frac{7}{3}|4.9\bar{3}\right)$ lies on the line $g_1$.
		\end{enumerate}
		
		{\bf \item Determine the equations:}
		
		\begin{enumerate}[a)]
			\item $g_4$: Parallel to the x-axis through $P(-1|5)$
			\item $g_5$: Angle bisector of the II and IV quadrants
			\item $g_5$: Line passing through $D(-3|1)$ and $B(7|-2)$
		\end{enumerate}
		
		{\bf \item Given is the line $g_6$ with the equation $ g_6: y = - \frac{3}{4}x + 2 $}
		
		\begin{enumerate}[a)]
			\item Calculate the intersection points $ S_y $ and $ S_x $ of the line $g_6$ with both coordinate axes.
			\item The point $ H(x_H|-\frac{1}{4})$ lies on the line $g_6$. Calculate the missing coordinate $x_H$
		\end{enumerate}
