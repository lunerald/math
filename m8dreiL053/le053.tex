	\begin{minipage}{0.6\textwidth}
		Construction of the angle bisector:
		\begin{itemize}
			\item Circle k(A;r) around A with arbitrary radius
			\item Mark the intersection points of k on the legs [AD] and [AC].
			\item Intersect circular arcs around the points
			\item Straight line through A and the intersection of the circular arcs = angle bisector
		\end{itemize}
		
		\noindent{\bf Task 1. b)}\\
		Construction of the rectangle:
		\begin{itemize}
			\item distance $\overline{DC}$ with the compass
			\item circular arc with this radius around A
		    \item distance $\overline{DA}$ with the compass
			\item circular arc with this radius around C
			\item intersect both arcs = B
			\item Draw route [AB] and [CB].
		\end{itemize}
		
		\noindent{\bf Task 1. c)}\\
		Construction of the mirror point D':
		\begin{itemize}
			\item Intersect circular arc around D with axis AC twice
			\item Circular arcs around the two intersection points
			\item Straight line through intersections of circular arcs = perpendicular from D to [AC]
			\item Length of point D -- Remove the plumb base point with the compass
			\item circular arc with this radius intersects around L with straight lines on the other side = D'
		\end{itemize}
