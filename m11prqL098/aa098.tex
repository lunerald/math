{\bf Ableiten mit der Produkt - und Quotientenregel}

\begin{enumerate}[1.]
{\bf\item Leiten Sie mit der Produktregel ab:}\\
{\it Terme müssen anschließend nicht vereinfacht werden}
$$(u(x)\cdot v(x))'=u'(x)v(x)+v'(x)u(x)$$
\renewcommand{\arraystretch}{3}
\begin{tabular}{p{5.3 cm}p{5.3 cm}p{5.3 cm}}
	a) $f(x)=(x^2+1)(x-2) $ & b) $f(x)=(x^3-2x)\cdot \sqrt{x} $ & c) $f(x)=(\sqrt{x}-1)\cdot 3x^5 $ \\
	 d) $f(x)=({1\over 2}x^4+{1\over 6}x^3)(tx-7) $ & e) $f(t)=(t^{1\over 2}+t^2)t^{-1} $ & f) $f(x)= (x-a)(x-b)$ \\
	  g) $f(x)=t^2x^2 \cdot (x-2t) $ & h) $f(t)=t^2x^2 \cdot (x-2t) $
\end{tabular}
{\bf\item Leiten Sie mit der kleinen Quotientenregel ab:}\\
$$\left({1 \over v(x)}\right)'={-v'(x)\over (v(x))^2}$$
\renewcommand{\arraystretch}{3}
\begin{tabular}{p{4 cm}p{4 cm}p{4 cm}p{4 cm}}
	a) $f(x)= {1\over x^2 }$  & b) $f(x)={1\over x^3 - 1 } $ & c) $f(x)=-{1\over x^2+x } $ & d) $f(x)=-{2\over x^2-2x } $ \\
    e) $f(x)={7 \over tx^3- t} $ & f) $f(t)={7 \over tx^3- t}$ & g) $f(x)= {a\over bx+c}$ & h) $f(t)=-{2\over t^2(t-3) } $
\end{tabular}
{\bf\item Leiten Sie mit der großen Quotientenregel ab:}\\
$$\left({u(x) \over v(x)}\right)'={u'(x)v(x)-v'(x)u(x)\over (v(x))^2}$$
\renewcommand{\arraystretch}{3}
\begin{tabular}{p{4 cm}p{4 cm}p{4 cm}p{4 cm}}
	a) $f(x)={x \over x+1} $ & b) $f(x)={-x^2 \over x-2}  $ & c) $f(x)={ax+b \over cx+d }  $ & d) $f(t)={ -t^3+2t +1  \over 4t^2}  $ \\ e) $f(x)={ t^2x\over x+t }  $ & f) $f(t)={ t^2x\over x+t } $ & g) $f(x)={ x(x+1)\over x-4}  $ & h) $f(t)={ -3t^4\over (t+1)^2} $ 
\end{tabular}
\end{enumerate}
