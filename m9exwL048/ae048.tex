		%Task 1
		{\bf \item Determine the extreme value of the given terms by calculation.} State the assignment of x for the extreme value and the type of extreme value.
		
		\begin{tabular}{cclcrcl}
			$ T_1(x) $ &=& $ x^2 + 3x + 1.25 $ & \parbox[20 pt][2em][c]{0 cm}{} &
			$ T_2(x) $ &=& $ - x^2 - 8x + 1 $ \\
			$ T_3(x) $ &=& $2x^2 -10x + 7.5 $ & \parbox[20 pt][2em][c]{0 cm}{} &
			$ T_4(x) $ &=& $- 3x^2 + 12x - 14 $ \\
			$ T_5(x) $ &=& $ - \frac{1}{9}x^2 + 2x - 3 $ & \parbox[20 pt][2em][c]{0 cm}{} &
			$ T_6(x) $ &=& $ - \frac{3}{5}x^2 - 2.4x + 2.1 $
		\end{tabular}
		
		%Exercise 2
		{\bf \item The extreme value of a quadratic term,} the type of extreme value and the assignment of x for the extreme value are known. Give a possible term with these properties.
		$$ T_{max} = - 8 \quad \textnormal{for} \quad x = 0.5 ; \quad T_7(x) = \cdots $$
		
		{\bf \item Give a term that has the minimum 2 for $ x = -3 $.} Put it in the form $ x^2 \pm bx \pm c $ !
		
		{\bf \item Determine the extreme value and the corresponding assignment for x !}
		
		\begin{enumerate}[a)]
			\item $ T(x) = x^2 - 5x + 8 $
			\item $ T(x) = 3 - (x + 1,3)^2$
			\item $ T(x) = x^2 - 11x$
			\item $ T(x) = 4 - x^2 $
		\end{enumerate}
		
		{\bf \item Specify the type and size of the extreme values:}
		
		\begin{tabular}{ccccc}
			a) & $T(x)$=& $0.5(x-0.5)^2 -3 $ &$T$\rule{20 pt}{0.5 pt}= \rule{30 pt}{0.5 pt }&
			for $x =$\rule{30 pt}{0.5 pt} \\
			b) & $T(x)$=&$-2(x+3)^2-1 $ &$T$\rule{20 pt}{0.5 pt} = \rule{30 pt}{0.5 pt}&
			for $x =$\rule{30 pt}{0.5 pt} \\
		\end{tabular}
		
		{\bf \item Reform so that you can determine the extreme values of the terms:}
		
		\begin{tabular}{cccccccc}
			a)&$T(x)$&=& $ x^2-2x+7$ & b) &$T(x)$&=&$x^2+9x+10$\\
			& &=&\rule{80 pt}{0.5 pt} & & &=& \rule{80 pt}{0.5 pt} \\
			& &=&\rule{80 pt}{0.5 pt} & & &=& \rule{80 pt}{0.5 pt}
		\end{tabular}
		
		{\bf \item The extreme value of a quadratic term,} the type of extremum and the associated assignment of x are known. Specify a possible term $T(x)$ with these properties:
		
		\begin{tabular}{cccccc}
			a) Extreme value:& $7$ & type: & maximum & location: $x=$& $-1$ \\
			b) Extreme value:& $-4$ & type: & minimum & location: $x=$& $3$
		\end{tabular}
		
		{\bf \item Calculate the extreme value of the term }
		$$ T(x) = 6x-0.25x^2-9x-3^2$$
