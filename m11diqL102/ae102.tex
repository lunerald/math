	{\bf Calculating tangents to function graphs}
	
	\begin{enumerate}[1.]
		\item Given the function $f(x)=-x^2+1; \qquad \mathbb{D}=\mathbb{R}$; its graph is $G_f$.
		Calculate an equation of the tangent $t$ to $G_f$ at the point
		
		\renewcommand{\arraystretch}{2}
		\begin{tabular}{cccc}
			a) $P(0|f(0))$ & b) $P(1|f(1))$ & c) $P(-2|f(-2))$ & d) $P(0 .5|f(0.5))$ \\
		\end{tabular}
		
		\item Given the function $f(x)={2\over x}; \qquad \mathbb{D}=\mathbb{R}\setminus \{0\}$; its graph is $G_f$.
		Calculate an equation of the tangent $t$ to $G_f$ at the point
		
		\renewcommand{\arraystretch}{2}
		\begin{tabular}{cccc}
			a) $P(1|f(1))$ & b) $P(-1|f(-1))$ & c) $P(2|f(2))$ & d) $P(0 .5|f(0.5))$ \\
		\end{tabular}
		
		\item Given the function $f(x)=x^3-1; \qquad \mathbb{D}=\mathbb{R}$; its graph is $G_f$.
		Calculate an equation of the tangent $t$ to $G_f$ at the point
		
		\renewcommand{\arraystretch}{2}
		\begin{tabular}{cccc}
			a) $P(1|f(1))$ & b) $P(-2|f(-2))$ & c) $P(0.5|f(0.5))$ & d) $P(0|f(0))$ \\
		\end{tabular}
		
		\item Given the function $f(x)={1\over 4}x^4; \qquad \mathbb{D}=\mathbb{R}$; its graph is $G_f$.
		Calculate an equation of the tangent $t$ to $G_f$ at the point
		
		\renewcommand{\arraystretch}{2}
		\begin{tabular}{cccc}
			a) $P(0|f(0))$ & b) $P(1|f(1))$ & c) $P(2|f(2))$ & d) $P(0.5 |f(0,5))$ \\
		\end{tabular}
		
		\vspace{1cm}
		
		\fbox{Instructions:}
		
		\fbox{
			
			\begin{minipage}{0.45\textwidth}
				With differential quotient:
				\begin{itemize}
					\item Determine y-coordinate of P ($=f(x_0$))
					\item Set up the differential quotient
					\item Transform and shorten differential quotient
					\item Limit is equal to tangent slope
					\item Set up the equation of the tangent $y=mx+t$
				\end{itemize}
		\end{minipage}}
		\hfill
		\fbox{
			\begin{minipage}{0.45\textwidth}
				With derivation function:
				\begin{itemize}
					\item Derive the function term
					\item Insert the x coordinate into the derivative function $f'(x)$\\ $\Rightarrow$ Tangent slope
					\item Determine y-coordinate of P ($=f(x_0$))
					\item Set up the equation of the tangent $y=mx+t$
				\end{itemize}
				
		\end{minipage}}
		
	\end{enumerate}
