	\noindent{\bf Task 1.}\\
	\begin{itemize}
		\item $\frac{e^2}{s}=g$: true (Catheter theorem)
		\item $ g^2 - e^2 = f^2 $: true (Pythagorean theorem)
		\item $\frac{h^2}{s}=r$: true (height set)
		\item $s^2=\sqrt{h^2-e^2}$: false; must be $s^2= e^2-h^2 $
	\end{itemize}

	\noindent{\bf Task 2.}\\
	The diagonal of the base area is:
	$$ e =\sqrt{230^2 + 230^2} = \SI{325}{\meter}$$
	Half of it is the line from one corner to the center. We apply the Pythagorean theorem:
	$$s = \sqrt{\left({e \over 2}\right)^2 + h^2}=\sqrt{163^2+140^2}=\SI{215}{\meter}$$
	The side edge is \SI{215}{\meter} long.

	\noindent{\bf Task 3.}\\
	The tree forms a right triangle with legs $a=5$; $b=x$ and the hypotenuse $c=25-x$. This means:
	$$
	\renewcommand{\arraystretch}{2}
	\begin{array}{rcll}
		(25-x)^2 &=& 5^2 + x^2 \\
		625 - 50x + x^2 &=& 25 + x^2 &|-x^2 + 50x -25\\
		600 &=& 50x \\
		x&=& \SI{12}{\meter}
	\end{array}
	$$
	The tree has snapped at a height of \SI{12}{\meter}.
	\vspace{0.5 cm}
	
	\noindent{\bf Task 4.}\\
	The longest side is the hypotenuse, so we set up:
	$$
	\renewcommand{\arraystretch}{2}
	\begin{array}{rcll}
		c^2 &=& a^2 + b^2 \\
		(4n^2+1)^2 &=& (4n^2-1)^2 + (4n)^2 \\
		16n^4 + 8n^2 +1 &=& 16n^4 - 8n^2 +1 + 16n^2\\
		0 &=& 0 &(w)
	\end{array}
	$$
	We get a true statement for all $n \in \mathbb{R}$. This means that the Pythagorean theorem is always fulfilled and the triangles are all right-angled.
	\vspace{0.5 cm}
	
	\noindent{\bf Task 5.}\\
	We form the right triangle $\Delta MDC$ with $\overline{MC}=r=\SI{8}{\centi\meter}$.
	This means:
	$$x=\overline{MD}=\sqrt{\overline{MC}^2-\overline{CD}^2}= \sqrt{8^2 - 4^2}=4\sqrt{3}=\SI{6.93}{\centi\meter}$$

	\noindent{\bf Task 6.}\\
	Let the width be $16x$ and the height be $9x$.
	Then we get:
	$$\renewcommand{\arraystretch}{2}
	\begin{array}{rcll}
		70^2 &=& (16x)^2 + (9x)^2\\
		4900 &=& 337x^2 &|:337\\
		14.5 &=& x^2\\
		3.8 &=& x \\
		&\Rightarrow& \textnormal{width} =16\cdot 3.8&=\SI{61}{\centi\meter}\\
		&\Rightarrow& \textnormal{height} =9\cdot 3.8&=\SI{34}{\centi\meter}\\
	\end{array}
	$$
