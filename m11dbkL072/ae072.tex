	{\bf continuity and differentiability}:\\
	
	\begin{enumerate}[1.]
		\item First examine the following functions at the interfaces for continuity and, if positive, then for differentiability there.
		
		
		\renewcommand{\arraystretch}{2}
		\begin{tabular}{clcl}
			a)&
			$
			f(x) =
			\left\{\begin{array}{ccl}
				{1\over 2} x +2 & \textnormal{for} & x \le 0 \\
				2 & \textnormal{for} & x > 0
			\end{array}\right.
			$
			&
			b)&
			$
			f(x) =
			\left\{\begin{array}{ccl}
				x^2 & \textnormal{for} & x < 1 \\
				2x-1 & \textnormal{for} & x \ge 1
			\end{array}\right.
			$
			\\
			&&&\\
			
			c)&
			$
			f(x) =
			\left\{\begin{array}{ccl}
				-x^3 +2x -4 & \textnormal{for} & x \le 2 \\
				7x^2 & \textnormal{for} & x > 2
			\end{array}\right.
			$
			&
			d)&
			$
			f(x) =
			\left\{\begin{array}{ccl}
				x^2 -4x +1 & \textnormal{for} & x \le 4 \\
				-(x-4)(x-8)+1& \textnormal{for} & x > 4
			\end{array}\right.
			$
			\\
			&&&\\
			e)&
			$
			f(x) =
			\left\{\begin{array}{ccl}
				4 & \textnormal{for} & x \le -1\\
				-0.5 x + 3 & \textnormal{for} & -1 < x \le 3 \\
				x^2 - 7.5 & \textnormal{for} & x > 3
			\end{array}\right.
			$
			&
			f)&
			$
			f(x) =
			\left\{\begin{array}{ccl}
				x^3 - 1 & \textnormal{for} & x \le 0 \\
				2x^2 - 1 & \textnormal{for} & 0 < x < 2 \\
				{1\over 2}x^4 -1 & \textnormal{for} & x \ge 2
			\end{array}\right.
			$
		\end{tabular}
		\vspace{1cm}
		
		\item Determine the parameter so that the functions at the interfaces are continuous. Then examine for differentiability
		
		\begin{tabular}{clcl}
			a)&
			$
			f(x) =
			\left\{\begin{array}{ccl}
				ax & \textnormal{for} & x \le 1 \\
				2x +1 & \textnormal{for} & 1 < x
			\end{array}\right.
			$
			&
			b)&
			$
			f(x) =
			\left\{\begin{array}{ccl}
				x + t & \textnormal{for} & x < -2 \\
				-{1\over 4}x^2& \textnormal{for} & x \ge -2
			\end{array}\right.
			$\\
			&&&\\
			c)&
			$
			f(x) =
			\left\{\begin{array}{ccl}
				- 3x +1 & \textnormal{for} & x\le b \\
				3x -1 & \textnormal{for} & x > b
			\end{array}\right.
			$
			&
			d)&
			$
			f(x) =
			\left\{\begin{array}{ccl}
				4x-4 & \textnormal{for} & x < c \\
				x^2 & \textnormal{for} & x \ge c
			\end{array}\right.
			$
		\end{tabular}
		\vspace{1cm}
		
		\item Investigate whether the following absolute value functions are not differentiable at one or more places and, if not, then specify these places.
		
		\begin{tabular}{p{3 cm}p{4 cm}p{3 cm}p{3 cm}}
			a) $f(x)=|x+5|$ & b) $f(x)=|x-t|$ & c) $f(x)=|{1\over 2}x-4|$ & d ) $f(x)=|-x+2|$ \\
			e) $f(x)=|x^2+1|$ & f) $f(x)=|(x+2)(x-3)|$ & g) $f(x)=|x^ 3|$ & h)$f(x)=|(x-1)^2|$
		\end{tabular}
		
	\end{enumerate}
