{\bf Stetigkeit und Differenzierbarkeit}:\\

\begin{enumerate}[1.]
\item Untersuchen Sie folgende Funktionen an den Nahtstellen zunächst auf Stetigkeit, und falls positiv, anschließend dort auf Differenzierbarkeit.


\renewcommand{\arraystretch}{2}
\begin{tabular}{clcl}
a)&
$
f(x) =
\left\{\begin{array}{ccl}
{1\over 2} x +2 & \textnormal{für} & x \le 0 \\
2 & \textnormal{für} & x > 0 
\end{array}\right.
$
&
b)&
$
f(x) =
\left\{\begin{array}{ccl}
x^2 & \textnormal{für} & x < 1 \\
2x-1 & \textnormal{für} & x \ge 1
\end{array}\right.
$
\\
&&&\\

c)&
$
f(x) =
\left\{\begin{array}{ccl}
-x^3 +2x -4 & \textnormal{für} & x \le 2 \\
7x^2 & \textnormal{für} & x > 2
\end{array}\right.
$
&
d)&
$
f(x) =
\left\{\begin{array}{ccl}
x^2 -4x +1 & \textnormal{für} & x \le 4 \\
-(x-4)(x-8)+1& \textnormal{für} & x > 4
\end{array}\right.
$
\\
&&&\\
e)&
$
f(x) =
\left\{\begin{array}{ccl}
4 & \textnormal{für} &  x \le -1\\
-0,5 x + 3 & \textnormal{für} & -1 < x \le 3 \\
x^2 - 7,5 & \textnormal{für} & x > 3
\end{array}\right.
$
&
f)&
$
f(x) =
\left\{\begin{array}{ccl}
x^3 - 1 & \textnormal{für} & x \le 0  \\
2x^2 - 1 & \textnormal{für} & 0 < x < 2 \\
{1\over 2}x^4 -1 & \textnormal{für} & x \ge 2
\end{array}\right.
$
\end{tabular}
\vspace{1 cm}

\item Bestimmen Sie den Parameter so, dass die Funktionen an den Nahtstellen stetig sind. Untersuchen Sie anschließend auf Differenzierbarkeit

\begin{tabular}{clcl}
a)&
$
f(x) =
\left\{\begin{array}{ccl}
 ax & \textnormal{für} & x \le 1 \\
 2x +1 & \textnormal{für} & 1 < x 
\end{array}\right.
$
&
b)&
$
f(x) =
\left\{\begin{array}{ccl}
x + t & \textnormal{für} & x < -2 \\
-{1\over 4}x^2& \textnormal{für} & x \ge -2
\end{array}\right.
$\\
&&&\\
c)&
$
f(x) =
\left\{\begin{array}{ccl}
- 3x +1 & \textnormal{für} & x\le b  \\
3x -1 & \textnormal{für} & x > b
\end{array}\right.
$
&
d)&
$
f(x) =
\left\{\begin{array}{ccl}
4x-4 & \textnormal{für} & x < c \\
x^2 & \textnormal{für} & x \ge c
\end{array}\right.
$
\end{tabular}
\vspace{1 cm}

\item Untersuchen Sie ob folgende Betragsfunktionen an einer oder mehreren Stellen nicht differenzierbar sind und, falls nicht, geben sie dann diese Stellen an.

\begin{tabular}{p{3 cm}p{4 cm}p{3 cm}p{3 cm}}
a) $f(x)=|x+5|$ & b) $f(x)=|x-t|$ & c) $f(x)=|{1\over 2}x-4|$ & d) $f(x)=|-x+2|$ \\
 e) $f(x)=|x^2+1|$ & f) $f(x)=|(x+2)(x-3)|$ & g) $f(x)=|x^3|$  & h)$f(x)=|(x-1)^2|$
\end{tabular}

\end{enumerate}
