	{\bf Training sheet for determining extreme values with the quadratic complement }\\
	\begin{enumerate}[1.]
		
		\begin{minipage}{10 cm}
			{\bf \item initial form $ T(x)= x^2 + px + q$:}\\
			$
			\renewcommand{\arraystretch}{1,5}
			\begin{array}{rcl}
				T(x)&=& x^2 + 3x + 6 \\
				&=& x^2 + 2\cdot 1.5x +6 \\
				&=& x^2 + 2\cdot 1.5x + 1.5^2 - 1.5^2 +6 \\
				&=& (x^2 + 2 \cdot 1.5x + 1.5^2) - 1.5^2 +6 \\
				&=& (x^2 + 2 \cdot 1.5x + 1.5^2)+ 3.75 \\
				&=& (x + 1.5)^2 + 3.75
			\end{array}
			$
			$$\Rightarrow T_{min}=3.75 \quad \textnormal{for} ~x=-1.5$$
			\vspace{35 pt}
			
			{\bf Carry out the extreme value determination using} \\
			quadratic complement after this\\
			method for the following terms:
			\begin{enumerate}[a)]
				\item $T(x)= x^2 -8x + $10
				\item $T(x)= x^2 + 6x +15 $
				\item $T(x)= x^2 -3x $
				\item $T(x)= x(x-4) $
				\item $T(x)= -10x +x^2 +30 $
				\item $T(x)= (x+3)(x-1)$
			\end{enumerate}
		\end{minipage}
		\hfill
		\begin{minipage}{10 cm}
			{\bf \item initial form $ T(x)= ax^2 + bx + c$:}\\
			$
			\renewcommand{\arraystretch}{1,5}
			\begin{array}{rcl}
				T(x)&=& -2x^2 + 12x - 7 \\
				&=& -2(x^2 - 6x)-7 \\
				&=& -2(x^2- 2\cdot 3x) -7\\
				&=& -2(x^2 - 2\cdot 3x + 3^2 - 3^2) -7 \\
				&=& -2(x^2 - 2 \cdot 3x + 3^2) +2 \cdot 3^2 -7 \\
				&=& -2(x^2 - 2 \cdot 3x + 3^2) + 11 \\
				&=& -2(x - 3)^2 + 11
			\end{array}
			$
			$$\Rightarrow T_{max}=11 \quad \textnormal{for} ~x=3$$
			
			{\bf Carry out the extreme value determination using} \\
			quadratic complement after this\\
			method for the following terms:
			\begin{enumerate}[a)]
				\item $T(x)= -x^2 -4x + 1 $
				\item $T(x)= 3x^2 + 18x + $42
				\item $T(x)=0.5x^2 -3x $
				\item $T(x)= -4x(3+x)+9 $
				\item $T(x)= {3 \over 2} x -{1 \over 4}x^2 -{9 \over 4} $
				\item $T(x)= (2x+4)(x-2)$
			\end{enumerate}
		\end{minipage}
		
	\end{enumerate}
