{\bf Grenzwerte}
\begin{enumerate}[1.]

%Aufgabe 1
{\bf \item Entscheide anhand des angegebenen Funktionstermes, ob die folgenden Grenzwerte existieren und gib gegebenenfalls den Grenzwert direkt an.}
 
\begin{tabular}{clccl}
a) & $\lim\limits_{x\to\infty}{4x^2- 9x^4}$&\parbox[20 pt][2em][c]{0 cm}{}& 
b) &  $\lim\limits_{x\to\infty}{(0,01x^3- 25x^2)}$\\
c) & $\lim\limits_{x\to-\infty}{\frac{8x+9}{2x-5}}$&\parbox[20 pt][2em][c]{0 cm}{}& 
d) &  $\lim\limits_{x\to\infty}{20 \cdot 2^{-x}}$\\
e) & $\lim\limits_{x\to-\infty}{\frac{\sin{x}}{x}}$&\parbox[20 pt][2em][c]{0 cm}{}& 
f) &  $\lim\limits_{x\to\infty}{\sqrt{x}\cdot \cos{2x}}$\\
g) & $\lim\limits_{x\to\infty}{\frac{x^3}{x^2+4}}$&\parbox[20 pt][2em][c]{0 cm}{}& 
h) &  $\lim\limits_{x\to-\infty}{\frac{x-10}{x^2-8}}$\\
\end{tabular}


{\bf \item Bestimme die folgenden Grenzwerte:}

\begin{enumerate}[a)]
\item $ \lim\limits_{x \to \infty} \frac{3x-5}{4x+3} $
\item $ \lim\limits_{x \to -\infty} \frac{4x+x^3}{x^4-2} $
\end{enumerate}

%Aufgabe 5
{\bf \item Berechne nachvollziehbar die folgenden Grenzwerte.}

\begin{tabular}{lll}
a) $\lim \limits_{x \to 0}{x^2-5 \over x+2}$ & b) $\lim \limits_{x \to 0^-}{5x^3 \over x^2-x}$  
& c)  $\lim \limits_{x \to 2^+}{1+x^2 \over x(x-2)}$\\
d) $\lim \limits_{x \to -\infty}{4x^2-3x^3 \over 2x^3-6}$ & e) 
$\lim \limits_{x \to 4}{3x-12 \over x^2+x-20}$ & f) 
 $\lim \limits_{x \to -\infty}{7x^2-2 \over 2x+3 }$ 
\end{tabular}

\end{enumerate}
