	\noindent{\bf Task 2.}\\
	
	$$\lim\limits_{x \to \infty} {3x-5 \over 4x+3}=
	\lim\limits_{x \to \infty} \frac{x(3 - {5 \over x})}{x(4 - {3\over x})}=
	{3 -0 \over 4 -0}={3 \over 4} $$
	$$\lim\limits_{x \to -\infty} {4x+x^3 \over x^4-2}=
	\lim\limits_{x \to -\infty} \frac{x^4({4\over x^3} + {1 \over x})}{x^4(1 - {2\over x^4 })}=
	{0 +0\over 1+0} =0$$
	
	\noindent{\bf Task 3. a)}\\
	Insert 0 directly:
	$$= -{5 \over 2} $$
	\noindent{\bf Task 3. b)}\\
	Exclude and reduce $x$ in the denominator, then insert 0:
	$$=\lim\limits_{x \to 0}{5x^2 \over x-1} = {0 \over -1} = 0 $$
	\noindent{\bf Task 3. c)}\\
	Insert 2 in the numerator, note the sign in the denominator for approximation:
	$$ "{1+4 \over 2 \cdot (+0)}"= + \infty$$
	\noindent{\bf Task 3. d)}\\
	numerator degree=denominator degree; Consider factors:
	$$= {-3 \over 2}$$
	\noindent{\bf Task 3. e)}\\
	Factor, reduce and insert:
	$$=\lim\limits_{x \to 4}{3(x-4)\over (x+5)(x-4)}=\lim\limits_{x \to 4} {3 \over x+ 5} = {1 \over 3}$$
	\noindent{\bf Task 3. f)}\\
	numerator degree $>$ denominator degree; Exclude, shorten, insert:
	$$= \lim\limits_{x \to -\infty}{x(7x-{2\over x})\over x(2+{3\over x})}= \lim\limits_{x \to -\infty}{7x-{2\over x}\over 2+{3\over x}}={-\infty \over 2}=-\infty $$
