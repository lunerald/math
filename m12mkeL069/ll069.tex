\noindent{\bf Aufgabe 1. e)}\\
Wir bestimmen die Nullstellen der Ableitung
$$
\renewcommand{\arraystretch}{2}
\begin{array}{rcll}
f'(x)&=& (x+1)\cdot e^x &\stackrel{!}{=}0\\
x&=&-1 
\end{array}
$$
Monotonietabelle:
$$
\renewcommand{\arraystretch}{2}
\begin{array}{c|c|c|c|c|c}

& x<-1 & x=-1 &  x>-1 \\
\hline
(x+1) & - & 0 & + \\
\hline
e^x & + & + & + \\
\hline
f'(x) & - & 0 & + \\
\hline
G_f & \searrow & \rightarrow & \nearrow\\
\hline
  & smf & T & sms
\end{array}
$$
$$f(-1)=-{1\over e} \quad \Rightarrow \quad \textnormal{Tiefpunkt} \quad T(-1|-{1\over e})$$
$G_f$ ist streng monoton fallend für $x \in ]-\infty;-1]$\\
$G_f$ ist streng monoton steigend für $x \in [-1;\infty[$

		\renewcommand{\arraystretch}{2}

		0&=&-{e^{-2}}\cdot x-{4e^{-2}} & \\

		\end{array}

		$$I_2=\int\limits_{-2}^{0} x\cdot e^x\textnormal{d}x = \left[(x-1)\cdot e^x\right]_{-2}^{0}=3e^{-2}-1$$
		Die Fläche ist dann:
		$$A=|I_1+I_2|=|e^{-1}-1|\approx 0,632~\textnormal{FE}$$
