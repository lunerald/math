	{\bf Mathematics -- Vector Calculus Q11}\\
	\begin{enumerate}[1.]
		
		%Task 1
		
		{\bf \item Specify two points P and Q so that their distance is 6 LE.} P and Q should be unequal to the coordinate origin.
		
		%Exercise 2
		
		{\bf \item Show that the vectors span a cube ABCDEFGH, i.e. that the vectors are the edges of a cube.}
		$$ \vec{AB} = \left(
		\begin{array}{c}
			{1}\\ {2}\\ {2}
		\end{array}
		\right) ;
		\hspace{15 pt}
		\vec{AD} = \left(
		\begin{array}{c}
			{2}\\ {1}\\ {-2}
		\end{array}
		\right) ;
		\hspace{15 pt}
		\vec{AE} = \left(
		\begin{array}{c}
			{2}\\ {-2}\\ {1}
		\end{array}
		\right) $$
		
		{\bf \item In a three-dimensional coordinate system with the origin O are the points
			$B(-3|4|0)$ and $D(-2|1|2)$ are given.}
		
		\begin{enumerate}[a)]
			\item Calculate the coordinates of the point $C$ so that the resulting quadrilateral is $OBCD$
			is a parallelogram.[To check $C(-5|5|2)$]
			\item The parallelogram $OBCD$ is the base of a straight pyramid with the apex $S$.
			Calculate the contents of the base of the pyramid.
			\item Determine the coordinates of $S$ so that the height of the pyramid is $5\sqrt{5}$ LE!
			\item The pyramid now rotates around the $x_2$ axis. The corner point $D$ moves on a circle. Give the coordinates of the center $N$ of this circle and calculate its radius!
		\end{enumerate}
		
		{\bf \item The points $A(4|0|3)$, $B(0|5|0)$, $C(1|5|7)$ and $D(-3|0|4) $ are vertices of a three-sided pyramid $ABCD$.}\
		\begin{enumerate}[a)]
			\item Draw an oblique image of the pyramid in a 3D coordinate system. How does the edge $[AD]$ run in the coordinate system?
			\item Show that the pyramid is a regular tetrahedron.
			\item The methane molecule $CH_4$ is built tetrahedral, with the $C$ atom at the spatial center $M_C$ and the four bonds to the $H$ atoms pointing to the tips. The coordinates of the spatial center are the arithmetic mean of the four corner points. Calculate the angle between two $C-H$ bonds.
			\item Calculate the volume and surface area of the tetrahedron.
			\item $M_1$, $M_2$, and $M_3$ are the midpoints of edges $[AB]$, $[AC]$ and $[AD]$. The pyramid $AM_1M_2M_3$ is separated from the pyramid $ABCD$ by a plane cut that runs through the center points. What is the volume and surface area of the small pyramid relative to the large one?[Hint:Central stretching]
			\item Calculate the volume and surface area of the truncated pyramid.
		\end{enumerate}
		{\bf \item Prove that in every parallelogram the sum of the squares over the four sides is as large as the sum of the squares over the two diagonals.}
		
		\begin{minipage}{0.7\textwidth}
			{\bf \item Project the triangle $ABC$ with $A(2|-3|7)$, $B(1|4|9)$ and $C(-3|-4|5)$ perpendicularly into the $x_1x_2$ plane.}
			\begin{enumerate}[a)]
				\item Specify the coordinates of points $A'$, $B'$ and $C'$.
				\item Compare the area of triangle $A'B'C'$ with that of triangle $ABC$.
			\end{enumerate}
		\end{minipage}
		\begin{minipage}{0.3\textwidth}
			\includegraphics[width= 5 cm]{proji337}
		\end{minipage}
		
		{\bf \item A ball with center $M(2|-1|7)$ passes through the point $A(8|2|5)$.}
		\begin{enumerate}[a)]
			\item Determine those points $B_k(5|5|k)$ with the real parameter $k$ that also lie on the surface of the sphere.
			\item Calculate the area of triangle $AB_9M$, where $B_9$ represents the point $(5|5|9)$.
		\end{enumerate}
		
	\end{enumerate}
