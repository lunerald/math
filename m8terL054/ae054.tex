	{\bf Area Terms}
	\begin{enumerate}[1.]
		
		%Task 1
		{\bf \item A rectangle has side lengths $\overline{AB}=\SI{12}{\centi\meter}$ and $\overline{AD}=\SI{5}{\centi\meter}$. } If you shorten $[AB]$ from $B$ by $x~\SI{}{\centi\meter}$ and at the same time extend $[AD]$ beyond $D$ by $x~\SI{} {\centi\meter}$, new rectangles $AB_nC_nD_n$ are created.
		
		\begin{enumerate}[a)]
			\item Draw the given rectangle and the rectangle $AB_1C_1D_1$ for $x=2$.
			\item Give a sensible base set for $x$.
			\item Calculate the perimeter of the rectangles $AB_nC_nD_n$ depending on $x$. What do you conclude from the result?
			\item Represent the area of the rectangles $AB_nC_nD_n$ depending on $x$.
			\item Calculate the maximum area and the corresponding $x$.
		\end{enumerate}
		
		%Exercise 2
		{\bf \item A rectangle $ABCD$ has side lengths $\overline{AB}=\SI{8}{\centi\meter}$ and $\overline{AD}=\SI{6}{\centi\meter }$.} If you extend $[AB]$ beyond $A$ and $B$ by {\bf each} $x~\SI{}{\centi\meter}$ and at the same time shorten $[AD]$ by $ x~\SI{}{\centi\meter}$, new rectangles are created.
		
		\begin{enumerate}[a)]
			\item Draw the original rectangle and the rectangle $A_1B_1C_1D_1$ for $x=1.5$.
			\item Give a sensible base set for $x$.
			\item Represent the perimeter of the rectangles $A_nB_nC_nD_n$ depending on $x$.
			\item Represent the area of the rectangles $A_nB_nC_nD_n$ depending on $x$.
			\item Calculate the maximum area and the corresponding $x$.
		\end{enumerate}
		
		%Task 3
		{\bf \item Given is a square $ABCD$ with side length $\SI{8}{\centi\meter}$.} If one extends one side of the square by $2x~\SI{}{\centi\meter} $ and at the same time shorten the other side by $x~\SI{}{\centi\meter}$, you get rectangles $AB_nC_nD_n$.
		
		\begin{enumerate}[a)]
			\item Draw the given square and calculate its perimeter and area.
			\item Complete the drawing of 3 a) with the rectangle $AB_1C_1D_1$ for $x=1$.
			\item Give a sensible base set for $x$.
			\item Represent the perimeter of the rectangles $AB_nC_nD_n$ depending on $x$.
			\item Represent the area of the rectangles $AB_nC_nD_n$ depending on $x$.
			\item Calculate the maximum area and the corresponding $x$.
		\end{enumerate}
		
		%Task 4
		{\bf \item An isosceles triangle $ABC$ has base length $\overline{AB}=\SI{4}{\centi\meter}$ and height $\SI{10}{\centi\meter}$. } If you extend $[AB]$ beyond $A$ and $B$ by {\bf each} $x~\SI{}{\centi\meter}$ and at the same time shorten the height by $x~\SI{} {\centi\meter}$, this creates new triangles $A_nB_nC_n$.
		
		\begin{enumerate}[a)]
			\item Draw the given triangle and calculate its area.
			\item Complete the drawing of 4 a) with the triangle $A_1B_1C_1$ for $x=2$.
			\item Give a sensible base set for $x$.
			\item Represent the area of the triangles $A_nB_nC_n$ depending on $x$.
			\item Calculate the maximum area and the corresponding $x$.
		\end{enumerate}
		
		%Task 4
		{\bf \item A rhombus $ABCD$ has one diagonal length $\overline{AC}=\SI{6}{\centi\meter}$ and the other diagonal length $\overline{BD}=\SI{8}{ \centi \meter}$.} If you extend $[AC]$ beyond $A$ and $C$ by {\bf each} $x~\SI{}{\centi \meter}$ and at the same time shorten $[BD ]$ on both sides by $x~\SI{}{\centi\meter}$, this creates new diamonds $A_nB_nC_nD_n$.
		
		\begin{enumerate}[a)]
			\item Draw the given rhombus and calculate its area.
			\item Complete the drawing in 5 a) with the rhombus $A_1B_1C_1D_1$ for $x=1$.
			\item Give a sensible base set for $x$.
			\item Represent the area of the diamonds $A_nB_nC_nD_n$ depending on $x$.
			\item Calculate the maximum area and the corresponding $x$. What special shape does this diamond have then?
		\end{enumerate}
		
		
	\end{enumerate}
