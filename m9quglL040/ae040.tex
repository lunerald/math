	{\bf Quadratic equations and parameter problems}
	\begin{enumerate}[1.]
		
		{\bf \item Add:}
		$$\frac{1}{2} \cdot (8x+ \dots)^2 = \dots x^2 + 8xy + \dots $$
		
		{\bf \item Calculate the solution set and represent it graphically on the number line:}
		$$ a) \quad -4x^2+2x \ge -2 \hspace{25 pt} b) \quad -\frac{1}{2}x^2+2x >2 $$
		
		{\bf \item Specify the number of solutions depending on $a\in \mathbb{R}$.}
		$$ ax^2-\frac{3}{2}x + 1=0$$
		
		{\bf \item For what slope $m$ is the line tangent to the parabola?}
		
		\begin{tabular}{ccl}
			$g:~y$&=&$mx+\frac{3}{2}$\\
			$p:~y$&=&$x^2-\frac{3}{2}x+\frac{3}{2}$
		\end{tabular}
		
		\noindent Use a simple freehand sketch to illustrate the position of the line and parabola for $m=1$.
		
		
		{\bf \item Given is the line} $$h: \quad y=\frac{9}{16}x-1.25 $$ and the family of lines
		$$g_a: -4y=-ax-5$$.
		\begin{enumerate}[a)]
			\item Show that the x-coordinates of the intersection points $S_a(x|y)$ of the line $h$ with the family of lines $g_a$ can be represented as follows depending on $a$:
			$$x= \frac{40}{9-4a}$$
			\item For which value of $a$ is there no intersection?
			\item Calculate the value of $a$ for $x=-8$.
		\end{enumerate}
		
		
		
	\end{enumerate}
