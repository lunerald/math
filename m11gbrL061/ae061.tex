	{\bf Fractional Rational Functions}
	\begin{enumerate}[1.]
		
		%Task 3
		{\bf \item Find the equation of a fractionally rational function that satisfies the following three conditions:}
		
		\begin{itemize}
			\item your graph intersects the x-axis at $x=2$,
			\item has a pole with no sign change at $x=3$ and
			\item $y=3$ is horizontal asymptote.
		\end{itemize}
		%Exercise 2
		{\bf \item The function is given}
		$f(x)= { 1 \over x} + 3 ; \quad D_f=\mathbb{R}\setminus \{0\}$;\\
		Using the differential quotient, calculate the derivative at the point $P(2|?)$.
		
		{\bf \item The function $f(x) = \frac{x^3}{x^2-4} $ is given. }
		
		\begin{enumerate}[a)]
			\item Determine the definition set $ D_f $ and indicate - with the help of a suitable calculation or justification - the type of definition gaps!
			\item Examine the function for symmetry.
			\item Determine the equations of all asymptotes.
			\item Show that the derivative has: $ f'(x) = \frac{x^4-12x^2}{{(x^2-4)}^2} $
			\item Determine the equation of the tangent to $ G_f$ at the point $ P (4|?) $. At what angle does the tangent intersect the x-axis?
		\end{enumerate}

		{\bf \item Given is the function $$f(x)= {x^2-9 \over 4x^4+4x^3-24x^2}$$}
		
		\begin{enumerate}[a)]
			\item Determine the maximum definition set and specify all zeros of $f$.
			\item Investigate mathematically or through appropriate considerations what type of definitional gaps are involved.
			\item Sketch $G_f$ using your previous results.
		\end{enumerate}
		
		%Task 3
		{\bf \item Given is the function $$ f(x)= 1- {2 \over x}+{1 \over x^2}$$}
		
		\begin{enumerate}[a)]
			\item Calculate the zeros of the function $f$ and write the equations of all asymptotes. Determine the behavior of $f$ in the neighborhood of $x=0$.
			\item Sketch the graph of the function.
		\end{enumerate}
		
		%Task 4
		{\bf \item Given is the function $$h(x)= {x^2+3x \over 4(x-1)}$$}
		
		\begin{enumerate}[a)]
			\item Determine the definition set as well as the type and position of the zeros of $h$.
			\item Perform all necessary limit considerations so that you can sketch the graph of $h$ with all of its asymptotes.
		\end{enumerate}
