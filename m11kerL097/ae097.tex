	{\bf Deriving with the chain rule}
	$$(u(v(x)))'=u'(v(x))\cdot v'(x)$$
	\begin{enumerate}[1.]
		{\bf \item Derive the following power functions once:}\\
		\renewcommand{\arraystretch}{3}
		\begin{tabular}{p{4 cm}p{4 cm}p{4 cm}p{4 cm}}
			a) $f(x)=(x^2-1)^7 $ & b) $f(x)={1 \over 2}\cdot(2x^3-x)^2 $ & c) $f (x)=(t^2x-4x)^3 $ & d) $f(t)=(t^2x-4x)^3 $\\
		\end{tabular}
		{\bf \item Derive the following trigonometric functions once:}\\
		\begin{tabular}{p{4 cm}p{4 cm}p{4 cm}p{4 cm}}
			a) $f(x)=\sin{2x} $ & b) $f(x)=\cos{(x^2-1)} $ & c) $f(x)=(\sin{x} )^9 $ & d) $f(x)={1\over 2}\cos{(4x-\pi)} $\\
		\end{tabular}
		{\bf \item Derive the following exponential functions once:}\\
		\begin{tabular}{p{4 cm}p{4 cm}p{4 cm}p{4 cm}}
			a) $f(x)= 2e^{-3x}$ & b) $f(x)=e^{-x^2+1} $ & c) $f(x)=e^{1\over x} $ & d) $f(x)=e^{ax+b} $\\
		\end{tabular}
		{\bf \item Derive the following logarithm functions once:}\\
		\begin{tabular}{p{4 cm}p{4 cm}p{4 cm}p{4 cm}}
			a) $f(x)=(\ln(x) -1)^2 $ & b) $f(x)=\ln {1\over x} $ & c) $f(t)=\ln( 2tx+5) $ & d) $f(x)=\ln(ax^2+bx+c) $\\
		\end{tabular}
		
		{\bf \item Derive the following irrational functions once:}\\
		\begin{tabular}{p{4 cm}p{4 cm}p{4 cm}p{4 cm}}
			a) $f(x)=\sqrt{2x-1} $ & b) $f(x)=\sqrt{t^2x^3+2t} $ & c) $f(t)=\sqrt{t ^2x^3+2t} $ & d) $f(x)=\sqrt[3]{(7x+1)^2} $\\
		\end{tabular}
		{\bf \item For experts: Derive:}\\
		a)$$f(x)=\tan(x)$$
		
		b) {\it 3-fold concatenation:}$$f(x)=(\sin(2x))^7$$
		
		c) {\it Tip: convert to e-function}$$f(x)=x^x$$
	\end{enumerate}
