		\item {\bf Differentiability}\\
		{\bf The function is given
			$$f(x)={1 \over |x-1|+1}$$}
		\begin{enumerate}
			\item Indicate at what point the function $f$ may not be differentiable.
			\item Show that the function $f$ is not differentiable at this point.
			\item Specify a function that is not differentiable at $x=-3$.
		\end{enumerate}
		\item {\bf Task} \\
		\includegraphics[width=10 cm]{nidiff246}
		\begin{enumerate}
			\item Specify all x-values where the graph has a slope of 0.
			\item Specify all x values at which the function is not differentiable.
		\end{enumerate}
		\item {\bf derivative function}\\
		The derivative function $f'$ of a function $f$ is the function that assigns the derivative $f'$ to every point x at which $f$ is differentiable:
		$$f'(x)=\lim\limits_{h\to 0} \frac{f(x+h)-f(x)}{h}$$
		\begin{enumerate}
			\begin{minipage}{0.5\textwidth}
				\item Given is the purely quadratic function $f(x)=0.25x^2$. Draw the tangents $t_{-3}$, $t_{-1}$, $t_0$, $t_1$ and $t_3$ to the graph of $f$.
				\item Calculate the derivative function $f'(x)$
				\item Calculate $f'(-3),\quad f'(-1), \quad f'(0), \quad f'(1), \quad f'(3)$
			\end{minipage}
			\begin{minipage}{0.5\textwidth}
				\includegraphics[width=6 cm]{reiq246}
			\end{minipage}
			
			\begin{minipage}{0.5\textwidth}
				\item Enter the points\\
				$A(-3|f'(-3)), \quad B(-1|f'(-1)), \quad C(0|f'(0))$\\
				$D(1|f'(1)), \quad E(3|f'(3))$ \\
				into the coordinate system.
				\item Enter the graph of the derivative function $f'$ into the coordinate system.
			\end{minipage}
			\begin{minipage}{0.5\textwidth}
				\includegraphics[width=6 cm]{leer246}
			\end{minipage}
		\end{enumerate}
