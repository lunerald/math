	\noindent {\bf Task 1. a)}\\
	The result space is (Euro returned = E; Drink received = G):
	$$\Omega= \{ E \cap G ; E \cap \bar{G}; \bar{E} \cap G ; \bar{E} \cap \bar{G} \}$$
	$$
	\renewcommand{\arraystretch}{2}
	\begin{array}{c|c|c|c}
		& E & \bar{E} & \\
		\hline
		G & {1\over 3} & {1\over 6}& {1\over 2} \\
		\hline
		\bar{G}& {1\over 6} & {1\over 3}& {1\over 2} \\
		\hline
		& {1\over 2} & {1\over 2}& 1
	\end{array}
	$$
	\noindent {\bf Exercise 1. b)}\\
	The desired probability is
	$$P(G\cap \bar{E})= {1 \over 6}$$
	\noindent {\bf Exercise 1. c)}\\
	The desired probability for this exclusive "or" is:
	$$P( G \cap \bar{E})+ P(\bar{G} \cap E) = {1 \over 6 } + {1 \over 6} = { 1 \over 3}$$
	\noindent{\bf Exercise 1. d)}\\
	The desired conditional probability according to Bayes' theorem is:
	$$P_{\bar{E}}(G)={ P(\bar{E}\cap G) \over P(\bar{E})}= { {1 \over 6} \over {1 \over 2}}= {1 \over 3} $$
	\noindent{\bf Exercise 1. e)}\\
	Analogous to d):
	$$P_G(E)={ P(G \cap E) \over P(G)}= { {1 \over 3} \over {1 \over 2}}= {2 \over 3} $$
	
	\noindent {\bf Task 2.}\\
	B: Image noise; T: Sound noise. \\
	$P(B)=0.1$ \\ $P_B(T)=0.7$\\ $P_{\overline{B}}(\overline{T})=0.95$ \\
	This is an inverse problem; we calculate it using Bayes' formula for conditional probability:
	
	$$ P_T(\overline{B})={ P(T \cap \overline{B}) \over P(T)} \qquad(*) $$
	
	We create a four-field table for the desired probabilities:
	$$
	\begin{array}{cccc}
		P(\overline{T} \cap \overline{B}) &= P(\overline{B}) \cdot P_{\overline{B}}(\overline{T})
		&= 0.9 \cdot 0.95&=0.855 \\
		P(T \cap B)&= P(B) \cdot P_B(T)&= 0.1 \cdot 0.7 &= 0.07
	\end{array}
	$$
	
	$$
	\renewcommand{\arraystretch}{1.5}
	\begin{array}{c|c|c|c}
		& B & \overline{B} & \\
		\hline
		T & 0.07 & 0.045 & 0.115 \\
		\hline
		\overline{T} & 0.03 & 0.855 & 0.885 \\
		\hline
		& 0,1 & 0.9 & 1.00
	\end{array}
	$$
	
	So, (*), the desired probability is:
	$$ P_T(\overline{B}) = {0.045 \over 0.115} = 0.3913 \approx 39\%$$

	\noindent {\bf Exercise 7}
	
	Since the events are independent:
	\[
	P(A\text{ even} \cap B\text{ even}) = P(A\text{ even}) \cdot P(B\text{ even}) = 0{.}70 \cdot 0{.}40 = \boxed{0{.}28}
	\]
	
	{\bf Four-field table:}
	
	\begin{center}
		\begin{tabular}{lcc|c}
			& \textbf{B even (0{.}40)} & \textbf{B odd (0{.}60)} & \textbf{sum} \\
			\midrule
			\textbf{A even (0{.}70)} & $0{.}28$ & $0{.}70 - 0{.}28 = 0{.}42$ & $0.70$ \\
			\textbf{A odd (0.30)} & $0.40 - 0.28 = 0.12$ & $0.30 - 0.12 = 0.18$ & $0.30$ \\
			\midrule
			\textbf{Sum} & $0.40$ & $0.60$ & $1.00$
		\end{tabular}
	\end{center}
	
	{\bf Probability: at least one die is even:}
	
	We use the opposite probability approach:
	
	\[
	P(\text{at least one even}) = 1 - P(\text{both odd}) = 1 - 0.18 = \boxed{0.82}
	\]
