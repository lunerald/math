%Aufgabe 1
\item {\bf Bilden Sie die 1. Ableitungen der folgenden Funktionen ohne weitere Vereinfachung:}
\begin{enumerate}[a)]
 \item $f(x)= \sqrt{5x^2+1}$
\item $g(x)=x \cdot \ln (9-x^2)$
\end{enumerate}
%Aufgabe 2
\item {\bf Gegeben ist die Funktion $f(x)=\ln (2x-3) $ mit maximaler Definitionsmenge $D_f$.}
\begin{enumerate}[a)]
\item Geben Sie den maximalen Definitionsbereich $D_f$ und die zugehörige Wertemenge $W_f$ an.
\item Bestimmen Sie die Gleichung den Tangente im Schnittpunkt des Graphen von $f$ mit der x-Achse.
\item Begründen Sie, dass $f(x)$ eine Umkehrfunktion besitzt. Bestimmen Sie deren Funktionsgleichung $y=f^{-1}(x)$ 
mit zugehöriger Definitionsmenge $D_{f^{-1}}$.
\end{enumerate}
%Aufgabe 3
\item {\bf Gegeben ist die Funktion $f(x)=x \cdot e^{1-x}$; $\,x \in R$}
\begin{enumerate}[a)]
\item Zeigen Sie, dass für die erste Ableitung von f gilt: $f'(x)=(1-x) \cdot e^{1-x}$
\item Bestimmen Sie das Monotonieverhalten des Graphen von $f$, sowie Lage und Art des Extrempunktes von $f$.
\item Zeigen Sie, dass $F(x)=-e^{1-x}-f(x)$ eine Stammfunktion von $f$ ist und bestimmen Sie die Nullstelle von $F(x)$.
\end{enumerate}

\end{enumerate}
