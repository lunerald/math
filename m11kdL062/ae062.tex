	\begin{enumerate}[1.]
		%Task 1
		\item {\bf Form the first derivatives of the following functions without further simplification:}
		\begin{enumerate}[a)]
			\item $f(x)= \sqrt{5x^2+1}$
			\item $g(x)=x \cdot \ln (9-x^2)$
		\end{enumerate}
		%Exercise 2
		\item {\bf Given is the function $f(x)=\ln (2x-3) $ with maximum definition set $D_f$.}
		\begin{enumerate}[a)]
			\item Specify the maximum domain $D_f$ and the associated value set $W_f$.
			\item Determine the equation of the tangent at the intersection of the graph of $f$ with the x-axis.
			\item Justify that $f(x)$ has an inverse function. Determine their functional equation $y=f^{-1}(x)$
			with associated definition set $D_{f^{-1}}$.
		\end{enumerate}
		%Task 3
		\item {\bf The function $f(x)=x \cdot e^{1-x}$ is given; $\,x \in R$}
		\begin{enumerate}[a)]
			\item Show that the first derivative of f has: $f'(x)=(1-x) \cdot e^{1-x}$
			\item Determine the monotonicity behavior of the graph of $f$, as well as the location and type of the extreme point of $f$.
			\item Show that $F(x)=-e^{1-x}-f(x)$ is an antiderivative of $f$ and find the zero of $F(x)$.
		\end{enumerate}
		
	\end{enumerate}
