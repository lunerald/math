\noindent{\bf Aufgabe 1. e)}\\
Wir bestimmen die Nullstellen der Ableitung
$$
\renewcommand{\arraystretch}{2}
\begin{array}{rcll}
f'(x)&=&-3x^2+6x&\stackrel{!}{=}0\\
0&=&x(-3x+6) &\Rightarrow \quad x_1=0 \\
0&=&-3x+6 &\Rightarrow \quad x_2=2
\end{array}
$$
Monotonietabelle:
$$
\renewcommand{\arraystretch}{2}
\begin{array}{c|c|c|c|c|c}

& x<0 & x=0 & 0<x<2 & x=2 & x>2 \\
\hline
x & - & 0 & + & + & +\\
\hline
-3x+6 & + & + & + & 0 & -\\
\hline
f'(x) & - & 0 & + & 0 & -\\
\hline
G_f & \searrow & \rightarrow & \nearrow & \rightarrow & \searrow \\
\hline
  & smf & T & sms & H & smf
\end{array}
$$
$$f(0)=0 \quad \Rightarrow \quad \textnormal{Tiefpunkt} \quad T(0|0)$$
$$f(2)=4 \quad \Rightarrow \quad \textnormal{Hochpunkt} \quad H(2|4)$$
$G_f$ ist streng monoton fallend für $x \in ]-\infty;0] \lor [2;\infty[$\\
$G_f$ ist streng monoton steigend für $x \in [0;2]$

\noindent{\bf Aufgabe 1. f)}\\
Wir bestimmen die Nullstellen der zweiten Ableitung
$$
\renewcommand{\arraystretch}{2}
\begin{array}{rcll}

f''(x)&=&-6x+6  &\stackrel{!}{=}0 \\
0&=& -6x+6 & \Rightarrow \quad x=1 
\end{array}
$$
Krümmungstabelle:
$$
\renewcommand{\arraystretch}{2}
\begin{array}{c|c|c|c}

& x<1 & x=1 & x>1 \\
\hline
-6x+6 & + & 0 & - \\
\hline
f''(x)& + & 0 & -\\
\hline
G_f & \curvearrowleft & \rightarrow & \curvearrowright\\
\hline
 & lk & W & rk
\end{array}
$$
$$f(1)=2 \quad \Rightarrow \quad \textnormal{Wendepunkt} \quad W(1|2)$$
$G_f$ ist linksgekrümmt für $x \in ]-\infty;1] $\\
$G_f$ ist rechtsgekrümmt für $x \in ]1;\infty[$
