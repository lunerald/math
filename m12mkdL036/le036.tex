	\noindent{\bf Task 1. e)}\\
	We determine the roots of the derivative
	$$
	\renewcommand{\arraystretch}{2}
	\begin{array}{rcll}
		f'(x)&=&-3x^2+6x&\stackrel{!}{=}0\\
		0&=&x(-3x+6) &\Rightarrow \quad x_1=0 \\
		0&=&-3x+6 &\Rightarrow \quad x_2=2
	\end{array}
	$$
	Monotonicity table:
	$$
	\renewcommand{\arraystretch}{2}
	\begin{array}{c|c|c|c|c|c}
		
		& x<0 & x=0 & 0<x<2 & x=2 & x>2 \\
		\hline
		x & - & 0 & + & + & +\\
		\hline
		-3x+6 & + & + & + & 0 & -\\
		\hline
		f'(x) & - & 0 & + & 0 & -\\
		\hline
		G_f & \searrow & \rightarrow & \nearrow & \rightarrow & \searrow \\
		\hline
		& smf & T & sms & H & smf
	\end{array}
	$$
	$$f(0)=0 \quad \Rightarrow \quad \textnormal{low point} \quad T(0|0)$$
	$$f(2)=4 \quad \Rightarrow \quad \textnormal{highpoint} \quad H(2|4)$$
	$G_f$ is strictly monotonically decreasing for $x \in ]-\infty;0] \lor [2;\infty[$\\
	$G_f$ is strictly monotonically increasing for $x \in [0;2]$

	\noindent{\bf Task 1. f)}\\
	We determine the roots of the second derivative
	$$
	\renewcommand{\arraystretch}{2}
	\begin{array}{rcll}
		
		f''(x)&=&-6x+6 &\stackrel{!}{=}0 \\
		0&=& -6x+6 & \Rightarrow \quad x=1
	\end{array}
	$$
	Curvature table:
	$$
	\renewcommand{\arraystretch}{2}
	\begin{array}{c|c|c|c}
		
		& x<1 & x=1 & x>1 \\
		\hline
		-6x+6 & + & 0 & - \\
		\hline
		f''(x)& + & 0 & -\\
		\hline
		G_f & \curvearrowleft & \rightarrow & \curvearrowright\\
		\hline
		& lk & W & rk
	\end{array}
	$$
	$$f(1)=2 \quad \Rightarrow \quad \textnormal{turning point} \quad W(1|2)$$
	$G_f$ is curved to the left for $x \in ]-\infty;1] $\\
	$G_f$ is right-curved for $x \in ]1;\infty[$
