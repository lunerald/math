	{\bf Fractional terms - for specialists}
	\begin{enumerate}[1.]
		
		{\bf \item The function f is given by $f(x) = \frac{1}{x - 2} - 1 $}
		
		Sketch the graph of the function in a suitable coordinate system and calculate the zero of the function $f$.
		
		{\bf \item Lunerald calculates:}
		$$\frac{x}{2x - 4} + \frac{1}{x - 2} = \frac{x(x-2)+(2x-4)}{(2x-4)(x-2 )}=
		\frac{x^2-4}{2x^2-8x+8}$$
		Evaluate this calculation, justify in particular whether Lunerald has already given the final result and carry out the calculation - if necessary in a different way!
		
		{\bf \item Solve the following fraction equations: }
		
		\renewcommand{\arraystretch}{3}
		\begin{tabular}{p{3.5 cm}p{3.5 cm}p{3.5 cm}p{3.5 cm}}
			a) ${1 \over x+1} = {3 \over x-1}$ & b) ${ 4\over y } = { 3\over y-1 }$ &c) ${1 \over x} =1- { 2 \over x}$ &d) ${4x \over x+1} = 4+ {1 \over x}$ \\
			e) ${x \over x+1} = {4 \over 2x+2}$ & f) ${2x \over x-1 } = -{6 \over 1-x}$ & g) ${x ^2 \over x^2+2x } = {5 \over x+2}$ &h) ${x-5 \over x } = {-2x^2 \over 3x^2 }$ \\
			i) ${x-3 \over 2x-6} = {5 \over x}$ & j) ${x \over 2x+8} = {5 \over x+4}+1$ &k) ${x +1 \over 4-2x} = {1 \over 2}+{1 \over x-2}$ &l) $\frac{1}{6x-2} = 1 + \frac{x}{9x-3 }$ \\
		\end{tabular}
		
		{\bf \item Simplify the following fraction terms} \quad {\it Don't forget to abbreviate!}
		
		\renewcommand{\arraystretch}{2}
		\begin{tabular}{p{3.5 cm}p{3.5 cm}p{3.5 cm}p{3.5 cm}}
			a) ${4 \over 2x} + {x-2 \over x} $ &b) ${x^2 \over x+1} - {1+2x^2 \over 2x+2 }$& c) $ {1 \over x+1}- {1 \over x+3}$ & d) ${-10 \over 5x+x^2} + {2 \over x}$\\
			e) ${x^2 +2x +1 \over x+1}$ &f) ${2x^2 +x \over x-x^2}$ &g) ${ x+3 \over 9-x^2}$ &h) ${x^2-8x \over x+1 } \cdot {3+3x \over x}$ \\
			i) ${{4 } \over (x+1)^2} : {2 \over 2x+2}$ &j) ${-4x + x^2 + 4 \over x} : {2-x \over 2x}$ &k) $\frac{5x}{x-2} \cdot \frac{4-2x}{x^2+2x}$ &l) $\left(\frac{1}{2x} + \frac {2}{x}\right) : \frac{3}{x-1} $
		\end{tabular}
		
		{\bf \item Simplify the following double fractions:} \quad {\it Don't forget to abbreviate!}
