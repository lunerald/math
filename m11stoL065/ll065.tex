\noindent{\bf Aufgabe 3.}\\
Es gilt:
$$P(B)=1-P(\overline{B})=1-{2 \over 3}= {1 \over 3}$$
$$
\renewcommand{\arraystretch}{1.5}
\begin{array}{rcll}
  E_1 &=&\{ A \cup B\}\\
P(E_1)&=& P(A)+P(B) - P(A \cap B) &\textnormal{Satz von Sylvester}\\
      &=&{3 \over 4} + {1 \over 3} - {1 \over 6}\\
      &=& {11 \over 12}\\
{\bf b)}\quad E_2 &=&\{ \overline{A} \cup \overline{B}\}= \{ \overline{A \cap B}\} 
  &\textnormal{De Morgan'sche Regel}\\
P(E_2)&=& 1-P(A \cap B) = 1-{1 \over 6}\\
      &=& 5 \over 6
\end{array}
$$
