	\noindent{\bf Cartesian coordinates and polar coordinates}\\
	
	\noindent
	\begin{minipage}{0.75\textwidth}
		Each point $P(x|y)$ of a Cartesian coordinate system can also be represented as a pair of values $P(r|\varphi)$ with the distance from the origin $r$ and the angle $\varphi$. \\
		It applies to $P(r|\varphi) \Rightarrow P(x|y)$:
		$$ x = r \cdot \cos \varphi \qquad ; \qquad y =r \cdot \sin \varphi$$
		and for $P(x|y) \Rightarrow P(r|\varphi)$:
		$$ r = \sqrt{x^2 + y^2} \qquad ; \qquad \varphi = \arctan{y \over x}$$
		
	\end{minipage}
